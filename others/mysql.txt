CREATE TABLE IF NOT EXISTS tcprtt (id INT PRIMARY KEY AUTO_INCREMENT,src_ip VARCHAR(50) NOT NULL,dest_ip VARCHAR(50) NOT NULL,	src_port VARCHAR(50) NOT NULL,dest_port VARCHAR(50) NOT NULL,rtt_time INT NOT NULL,createAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,updateAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP);

CREATE TABLE IF NOT EXISTS tcpretrans (id INT PRIMARY KEY AUTO_INCREMENT,pid VARCHAR(50) NOT NULL,ip VARCHAR(50) NOT NULL, src_ip VARCHAR(50) NOT NULL,dest_ip VARCHAR(50) NOT NULL,	src_port VARCHAR(50) NOT NULL,dest_port VARCHAR(50) NOT NULL,createAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,updateAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP);

CREATE TABLE IF NOT EXISTS tcpconnlat (id INT PRIMARY KEY AUTO_INCREMENT,pid VARCHAR(50) NOT NULL,ip VARCHAR(50) NOT NULL, src_ip VARCHAR(50) NOT NULL,dest_ip VARCHAR(50) NOT NULL,	src_port VARCHAR(50) NOT NULL,dest_port VARCHAR(50) NOT NULL,lat_time INT NOT NULL,createAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,updateAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP);

CREATE TABLE IF NOT EXISTS biolatency (id INT PRIMARY KEY AUTO_INCREMENT,latency INT NOT NULL,createAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,updateAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP);
